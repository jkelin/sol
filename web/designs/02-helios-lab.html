import { $component } from "solix";

// Setup runs once per mounted instance.
export const Counter = $component(() => {
  let count = 0;
  const doubled = count * 2;

  return (
    <section>
      <p>Solar count: {count}</p>
      <output>Energy: {doubled}</output>
      <button onClick={() => count++}>
        Increase
      </button>
    </section>
  );
});
